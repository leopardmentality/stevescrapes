def scrape_page(url: str) -> tuple[list[dict], str | None]:
    r = requests.get(url, headers=HEADERS, timeout=10)
    r.raise_for_status()
    soup = BeautifulSoup(r.text, "html.parser")

    books = []
    for art in soup.select("article.product_pod"):
        title       = art.h3.a["title"]
        price       = art.select_one("p.price_color").text.strip()
        rating_word = art.p["class"][1]          # e.g. "Three"
        stars       = RATING_MAP.get(rating_word, 0)
        img_url     = resolve_img(art.img["src"])
        link        = resolve_link(art.h3.a["href"])
        books.append({"title": title, "price": price,
                       "stars": stars, "img_url": img_url, "link": link})

    next_btn = soup.select_one("li.next a")
    next_url = resolve_next(url, next_btn["href"]) if next_btn else None
    return books, next_url
